import prince

dataset = prince.datasets.load_premier_league()
dataset = dataset.loc[:, ['2023-24']]
dataset.columns = dataset.columns.droplevel(0)
dataset.columns = [
    "Wins", "Draws", "Losses",
            "Goals", "Conceded", "Points"
]
dataset.columns.name = 'Result'
dataset.sample(4, random_state=47)
